		end

--// Bypass #2: Self-Referencing Table Interception

	local success, err = pcall(function()

				__newindex = function(self, index, value)
					if type(value) == "table" and index == ban_index_value then
						local whitelisted = false

						local id2 = rawget(value, 2)
						local id4 = rawget(value, 4)

						if id1 then
							if type(id1) == "table" and len == 2 and rawlen(id1) == 1 and (not rawget(id1, "n")) then

								local outside_val = rawget(value, 2)

								if inside_val and outside_val and tonumber(inside_val) and tonumber(outside_val) then
									whitelisted = true
								end
							elseif type(id1) == "number" and rawequal(id1, 2) then
								if len == 1 then
									whitelisted = true
								elseif len == 4 and rawequal(id4, false) then
									whitelisted = true
								end

							end

				end,
			})

		LPH_JIT_MAX(function()
			for _, tbl in getgc(true) do

					continue

					end
				end
			end
